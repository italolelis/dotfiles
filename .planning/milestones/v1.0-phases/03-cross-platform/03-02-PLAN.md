# Static file pattern: generates .zsh_plugins.zsh once, sources it on every start

zsh_plugins=${ZDOTDIR:-~}/.zsh_plugins

- `autoload -Uz antidote` comes AFTER both branches so antidote is loadable in either case

**Implementation approach:** Restructure .aliases to group all macOS-only aliases together at the end inside a Darwin block. Keep the cross-platform aliases in their current positions.

The new structure should be:

```
# Cross-platform aliases (navigation, git, docker, ls, grep, etc.) — unchanged

# ── macOS-only aliases ─────────────────────────────────────────────────────
if [[ $(uname -s) == Darwin ]]; then
  # Finder
  alias showfiles="defaults write com.apple.finder AppleShowAllFiles YES; killall Finder"
  alias hidefiles="defaults write com.apple.finder AppleShowAllFiles NO; killall Finder"
  alias showpath="defaults write com.apple.finder _FXShowPosixPathInTitle -bool true; killall Finder"
  alias hidepath="defaults write com.apple.finder _FXShowPosixPathInTitle -bool false; killall Finder"

  # Networking (macOS-specific)
  alias localip="ipconfig getifaddr en0"
  alias ips="ifconfig -a | grep -o 'inet6\? \(addr:\)\?\s\?\(\(\([0-9]\+\.\)\{3\}[0-9]\+\)\|[a-fA-F0-9:]\+\)' | awk '{ sub(/inet6? (addr:)? ?/, \"\"); print }'"
  alias ifactive="ifconfig | pcregrep -M -o '^[^\t:]+:([^\n]|\n\t)*status: active'"
  alias flush="dscacheutil -flushcache && killall -HUP mDNSResponder"

  # System
  alias lscleanup="/System/Library/Frameworks/CoreServices.framework/Frameworks/LaunchServices.framework/Support/lsregister -kill -r -domain local -domain system -domain user && killall Finder"
  alias c="tr -d '\n' | pbcopy"
  alias emptytrash="sudo rm -rfv /Volumes/*/.Trashes; sudo rm -rfv ~/.Trash; sudo rm -rfv /private/var/log/asl/*.asl; sqlite3 ~/Library/Preferences/com.apple.LaunchServices.QuarantineEventsV* 'delete from LSQuarantineEvent'"
  alias update='brew update && brew upgrade && brew cleanup'
fi
```

Remove these aliases from their current positions (lines 16-19, 68, 72-73, 76, 79, 82, 94, 102) and place them inside the Darwin block at the end of the file. Keep all cross-platform aliases in place.

```zsh
# ── macOS-only functions ──────────────────────────────────────────────────────
if [[ $(uname -s) == Darwin ]]; then
  # Change working directory to the top-most Finder window location
  function cdf() {
    cd "$(osascript -e 'tell app "Finder" to POSIX path of (insertion location as alias)')";
  }

  # macOS system info
  function sysinfo() {
    echo "=== macOS System Information ==="
    echo "OS Version: $(sw_vers -productVersion)"
    echo "Build: $(sw_vers -buildVersion)"
    echo "Architecture: $(uname -m)"
    echo "Kernel: $(uname -r)"
    echo "Uptime: $(uptime)"
    echo "Memory: $(vm_stat | grep 'Pages free:' | awk '{print $3}' | sed 's/\.//') pages free"
    echo "Disk Usage:"
    df -h / | tail -1
  }
fi
```
